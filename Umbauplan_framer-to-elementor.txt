// src/orchestrator/state-manager.ts
import { EventEmitter } from 'events';
import * as fs from 'fs/promises';
import * as path from 'path';

export interface PipelineState {
  currentPhase: number;
  completedPhases: number[];
  artifacts: {
    [key: string]: {
      path: string;
      hash: string;
      timestamp: string;
    };
  };
  variables: {
    [key: string]: any;
  };
  timestamp: string;
}

export class StateManager extends EventEmitter {
  private state: PipelineState;
  private statePath: string;

  constructor(statePath: string = './state.json') {
    super();
    this.statePath = statePath;
    this.state = this.initializeState();
  }

  private initializeState(): PipelineState {
    return {
      currentPhase: 0,
      completedPhases: [],
      artifacts: {},
      variables: {},
      timestamp: new Date().toISOString(),
    };
  }

  async saveState(): Promise<void> {
    this.state.timestamp = new Date().toISOString();
    await fs.writeFile(
      this.statePath,
      JSON.stringify(this.state, null, 2),
      'utf-8'
    );
    this.emit('stateSaved', this.state);
  }

  async loadState(): Promise<PipelineState | null> {
    try {
      const data = await fs.readFile(this.statePath, 'utf-8');
      this.state = JSON.parse(data);
      return this.state;
    } catch (error) {
      return null;
    }
  }

  async updatePhase(phase: number, completed: boolean = false): Promise<void> {
    this.state.currentPhase = phase;
    if (completed && !this.state.completedPhases.includes(phase)) {
      this.state.completedPhases.push(phase);
    }
    await this.saveState();
    this.emit('phaseUpdated', phase, completed);
  }

  async addArtifact(key: string, path: string, content: any): Promise<void> {
    const hash = await this.calculateHash(content);
    this.state.artifacts[key] = {
      path,
      hash,
      timestamp: new Date().toISOString(),
    };
    await this.saveState();
    this.emit('artifactAdded', key, path);
  }

  async verifyArtifactIntegrity(): Promise<boolean> {
    for (const [key, artifact] of Object.entries(this.state.artifacts)) {
      try {
        const content = await fs.readFile(artifact.path, 'utf-8');
        const currentHash = await this.calculateHash(content);
        if (currentHash !== artifact.hash) {
          this.emit('artifactCorrupted', key);
          return false;
        }
      } catch (error) {
        this.emit('artifactMissing', key);
        return false;
      }
    }
    return true;
  }

  private async calculateHash(content: any): Promise<string> {
    const crypto = await import('crypto');
    return crypto
      .createHash('sha256')
      .update(JSON.stringify(content))
      .digest('hex');
  }
}

// src/builder/mcp-adapter.ts
import { CircuitBreaker } from '../utils/circuit-breaker';
import { BatchScheduler } from '../utils/batch-scheduler';
import { IdempotencyKeyGenerator } from '../utils/idempotency';

export class NovamiraMcpAdapter {
  private circuitBreaker: CircuitBreaker;
  private batchScheduler: BatchScheduler;
  private keyGenerator: IdempotencyKeyGenerator;

  constructor(private config: McpConfig) {
    this.circuitBreaker = new CircuitBreaker({
      failureThreshold: 5,
      resetTimeout: 60000, // 1 minute
      monitor: (state) => this.handleCircuitStateChange(state),
    });

    this.batchScheduler = new BatchScheduler({
      maxBatchSize: 50,
      maxWaitTime: 5000, // 5 seconds
    });

    this.keyGenerator = new IdempotencyKeyGenerator();
  }

  async call<T = any>(
    abilityName: string,
    parameters: any,
    options: {
      idempotent?: boolean;
      timeout?: number;
      retries?: number;
    } = {}
  ): Promise<T> {
    const { idempotent = false, timeout = 30000, retries = 3 } = options;

    const idempotencyKey = idempotent
      ? this.keyGenerator.generateForParams(parameters)
      : this.keyGenerator.generateUnique();

    const call = () =>
      this.executeMcpCall(abilityName, parameters, {
        idempotencyKey,
        timeout,
      });

    return this.circuitBreaker.execute(call, { retries });
  }

  async batchCall(
    calls: Array<{ abilityName: string; parameters: any }>,
    options: { maxConcurrent?: number } = {}
  ): Promise<any[]> {
    const { maxConcurrent = 5 } = options;

    return this.batchScheduler.execute(
      calls.map((call) => () =>
        this.call(call.abilityName, call.parameters, { idempotent: true })
      ),
      { maxConcurrent }
    );
  }

  private async executeMcpCall(
    abilityName: string,
    parameters: any,
    options: { idempotencyKey: string; timeout: number }
  ): Promise<any> {
    const { idempotencyKey, timeout } = options;

    // MCP Call Implementation
    const response = await fetch(`${this.config.endpoint}/mcp/call`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Basic ${Buffer.from(
          `${this.config.username}:${this.config.password}`
        ).toString('base64')}`,
        'X-Idempotency-Key': idempotencyKey,
      },
      body: JSON.stringify({
        ability_name: abilityName,
        parameters,
      }),
      signal: AbortSignal.timeout(timeout),
    });

    if (!response.ok) {
      throw new McpError(
        `MCP call failed: ${response.statusText}`,
        response.status,
        abilityName
      );
    }

    return response.json();
  }

  private handleCircuitStateChange(state: 'closed' | 'open' | 'half-open'): void {
    console.warn(`Circuit breaker state changed: ${state}`);
    // Emit event, update monitoring, etc.
  }
}

class McpError extends Error {
  constructor(
    message: string,
    public statusCode: number,
    public abilityName: string
  ) {
    super(message);
    this.name = 'McpError';
  }
}

# Framer to Elementor V4 Pipeline Skill

## When to use this skill
- Converting Framer designs to Elementor V4
- Building atomic widget trees from Framer XML
- Validating V4 tree structures
- Managing design token extraction

## Prerequisites
- Node.js ≥ 18
- Unframer MCP access
- Novamira MCP server on WordPress
- FramerExport CLI

## Pipeline Phases
| Phase | Description | Command |
|-------|-------------|---------|
| 0 | MCP Check & Preflight | `npm run preflight` |
| 1 | Framer Export | `npm run export` |
| 2 | Token Extraction | `npm run token-extract` |
| 3 | Variable Creation | `npm run create-variables` |
| 4 | Conversion & GC | `npm run convert && npm run gc-generate` |
| 5 | Validation | `npm run validate` |
| 6 | Cross-Validation | `npm run cross-validate` |
| 7 | WordPress Build | `npm run build` |
| 8 | Post-Build QA | `npm run qa` |

## Critical Invariants
1. **Style-Classes Binding**: Every ID in `element.styles` MUST exist in `settings.classes.value`
2. **Visual Props Isolation**: Visual properties NEVER in `settings` - only in `styles`
3. **Style ID Format**: Only `[a-z][a-z0-9_]*` - NO hyphens
4. **Image Source Format**: `image-src` has either `id` OR `url`, never both
5. **Custom CSS Format**: Always `{"raw": "..."}` - never plain string

## Common Issues & Fixes
| Issue | Cause | Fix |
|-------|-------|-----|
| `class_name_contains_spaces` | Hyphen in style ID | Rename style ID (no hyphens) |
| `STYLE_CLASSES_BINDING FAIL` | Style ID missing in classes.value | Add style ID to `settings.classes.value` |
| Image doesn't load | `url: null` in image-src | Remove `url` key completely |
| Wrong colors | GV-ID drift after kit update | `adrians-export-design-system` → `cross-validate --design-system` |
| `elementor-set-content` timeout | Tree too large | `npm run dependency-graph` → build section-wise |

Every run saves a `state.json` in the research directory. Resume from any point:
```bash
npm run resume -- --state ./research/example.com/state.json
```

## Validation Checks
- **Score Threshold**: ≥ 85% required for build to proceed
- **14 Guards**: Pre-build validation including token existence, color consistency, font resolution
- **7 Cross-Validation Checks**: Including GV_ID_DRIFT detection
- **Schema Validation**: 5 checks including `$$type` correctness

// src/validator/guards/style-classes-binding.ts
import { GuardResult, GuardContext } from '../types';

export class StyleClassesBindingGuard {
  readonly id = 'STYLE_CLASSES_BINDING';
  readonly severity = 'error' as const;
  readonly description = 'Every style ID must be referenced in settings.classes.value';

  validate(context: GuardContext): GuardResult {
    const { tree, reportWarning, reportError } = context;
    const errors: string[] = [];
    const warnings: string[] = [];

    const traverse = (node: any, path: string = '') => {
      if (!node || typeof node !== 'object') return;

      const currentPath = path ? `${path}.${node.id || 'unknown'}` : 'root';

      // Check style-classes binding
      if (node.styles && typeof node.styles === 'object') {
        const styleIds = Object.keys(node.styles);
        const classesValue = node.settings?.classes?.value || [];

        for (const styleId of styleIds) {
          if (!classesValue.includes(styleId)) {
            errors.push(
              `Style ID "${styleId}" not found in settings.classes.value ` +
              `at ${currentPath}. Add it to the classes array.`
            );
          }
        }

        // Check for orphaned class references
        for (const classRef of classesValue) {
          if (!styleIds.includes(classRef) && !classRef.startsWith('gc-')) {
            warnings.push(
              `Class reference "${classRef}" not defined in styles ` +
              `at ${currentPath}. Consider removing it or defining the style.`
            );
          }
        }
      }

      // Recursively check children
      if (node.elements && Array.isArray(node.elements)) {
        node.elements.forEach((child: any, index: number) => {
          traverse(child, `${currentPath}.elements[${index}]`);
        });
      }
    };

    traverse(tree);

    return {
      guardId: this.id,
      severity: this.severity,
      errors,
      warnings,
      score: errors.length === 0 ? 100 : Math.max(0, 100 - errors.length * 10),
    };
  }
}

<summary>🔧 Abhängigkeitsgraph-Builder</summary>

```typescript
// src/orchestrator/dependency-graph.ts
import { EventEmitter } from 'events';

export interface DependencyNode {
  id: string;
  type: 'section' | 'widget' | 'style' | 'variable';
  dependencies: string[];
  dependents: string[];
  data: any;
}

export class DependencyGraphBuilder extends EventEmitter {
  private nodes: Map<string, DependencyNode> = new Map();

  addNode(id: string, type: DependencyNode['type'], data: any): void {
    if (this.nodes.has(id)) {
      this.emit('warning', `Node ${id} already exists, overwriting`);
    }

    this.nodes.set(id, {
      id,
      type,
      dependencies: [],
      dependents: [],
      data,
    });
    this.emit('nodeAdded', id, type);
  }

  addDependency(from: string, to: string): void {
    const fromNode = this.nodes.get(from);
    const toNode = this.nodes.get(to);

    if (!fromNode || !toNode) {
      throw new Error(`Cannot add dependency: missing node(s)`);
    }

    if (!fromNode.dependencies.includes(to)) {
      fromNode.dependencies.push(to);
    }

    if (!toNode.dependents.includes(from)) {
      toNode.dependents.push(from);
    }

    this.emit('dependencyAdded', from, to);
  }

  calculateBuildOrder(): string[] {
    const visited = new Set<string>();
    const visiting = new Set<string>();
    const order: string[] = [];

    const visit = (nodeId: string) => {
      if (visited.has(nodeId)) return;
      if (visiting.has(nodeId)) {
        throw new Error(`Circular dependency detected at ${nodeId}`);
      }

      visiting.add(nodeId);
      const node = this.nodes.get(nodeId);

      if (node) {
        for (const dep of node.dependencies) {
          visit(dep);
        }
        visiting.delete(nodeId);
        visited.add(nodeId);
        order.push(nodeId);
      }
    };

    for (const nodeId of this.nodes.keys()) {
      visit(nodeId);
    }

    return order;
  }

  detectCircularDependencies(): string[][] {
    const cycles: string[][] = [];
    const visited = new Set<string>();
    const recursionStack = new Set<string>();

    const dfs = (nodeId: string, path: string[]): void => {
      visited.add(nodeId);
      recursionStack.add(nodeId);

      const node = this.nodes.get(nodeId);
      if (node) {
        for (const dep of node.dependencies) {
          if (!visited.has(dep)) {
            dfs(dep, [...path, dep]);
          } else if (recursionStack.has(dep)) {
            // Found cycle
            const cycleStart = path.indexOf(dep);
            cycles.push([...path.slice(cycleStart), dep]);
          }
        }
      }

      recursionStack.delete(nodeId);
    };

    for (const nodeId of this.nodes.keys()) {
      if (!visited.has(nodeId)) {
        dfs(nodeId, [nodeId]);
      }
    }

    return cycles;
  }

  getStats(): {
    totalNodes: number;
    nodesByType: Record<string, number>;
    maxDepth: number;
    avgDependencies: number;
  } {
    const stats = {
      totalNodes: this.nodes.size,
      nodesByType: {} as Record<string, number>,
      maxDepth: 0,
      avgDependencies: 0,
    };

    let totalDeps = 0;
    for (const node of this.nodes.values()) {
      stats.nodesByType[node.type] = (stats.nodesByType[node.type] || 0) + 1;
      totalDeps += node.dependencies.length;
    }

    stats.avgDependencies = totalDeps / Math.max(1, stats.totalNodes);

    // Calculate max depth
    const visited = new Set<string>();
    const calculateDepth = (nodeId: string, depth: number = 0): number => {
      if (visited.has(nodeId)) return 0;
      visited.add(nodeId);

      const node = this.nodes.get(nodeId);
      if (!node || node.dependencies.length === 0) return depth;

      return Math.max(
        ...node.dependencies.map((dep) => calculateDepth(dep, depth + 1))
      );
    };

    for (const nodeId of this.nodes.keys()) {
      const depth = calculateDepth(nodeId);
      stats.maxDepth = Math.max(stats.maxDepth, depth);
    }

    return stats;
  }
}

// src/extractor/v4-atomic-detector.ts
import { PlaywrightBrowser } from '../scraper/browser';
import { V4WidgetType } from '../types/v4-types';

export class V4AtomicDetector {
  private browser: PlaywrightBrowser;

  constructor(browser: PlaywrightBrowser) {
    this.browser = browser;
  }

  async detectAtomicWidgets(url: string): Promise<V4WidgetDetection[]> {
    const page = await this.browser.newPage();
    await page.goto(url, { waitUntil: 'networkidle' });

    const detections = await page.evaluate(() => {
      const widgets: V4WidgetDetection[] = [];
      const observer = new MutationObserver((mutations) => {
        for (const mutation of mutations) {
          for (const node of mutation.addedNodes) {
            if (node instanceof HTMLElement) {
              this.checkElementForAtomicWidget(node, widgets);
            }
          }
        }
      });

      observer.observe(document.body, {
        childList: true,
        subtree: true,
      });

      // Check existing elements
      const allElements = document.querySelectorAll('*');
      allElements.forEach((el) => {
        this.checkElementForAtomicWidget(el, widgets);
      });

      return widgets;
    });

    await page.close();
    return detections;
  }

  private checkElementForAtomicWidget(
    element: HTMLElement,
    widgets: V4WidgetDetection[]
  ): void {
    // Check for V4 atomic widget indicators
    const widgetType = this.detectWidgetType(element);
    if (widgetType) {
      widgets.push({
        element: element.outerHTML,
        type: widgetType,
        classes: Array.from(element.classList),
        styles: this.extractInlineStyles(element),
        children: this.countChildren(element),
      });
    }
  }

  private detectWidgetType(element: HTMLElement): V4WidgetType | null {
    // V4 atomic widget detection logic
    const className = element.className;
    const dataAttributes = element.dataset;

    // Check for common V4 widget patterns
    if (className.includes('e-flexbox')) return 'e-flexbox';
    if (className.includes('e-div-block')) return 'e-div-block';
    if (className.includes('e-heading')) return 'e-heading';
    if (className.includes('e-button')) return 'e-button';
    if (className.includes('e-image')) return 'e-image';
    if (className.includes('e-svg')) return 'e-svg';

    // Check for widget-specific data attributes
    if (dataAttributes.widgetType) {
      return dataAttributes.widgetType as V4WidgetType;
    }

    return null;
  }

  private extractInlineStyles(element: HTMLElement): Record<string, string> {
    const styles: Record<string, string> = {};
    const computed = window.getComputedStyle(element);

    // Extract relevant V4 style properties
    const relevantProps = [
      'width',
      'height',
      'padding',
      'margin',
      'background-color',
      'color',
      'font-family',
      'font-size',
      'font-weight',
      'line-height',
      'text-align',
      'display',
      'flex-direction',
      'justify-content',
      'align-items',
      'gap',
    ];

    for (const prop of relevantProps) {
      styles[prop] = computed.getPropertyValue(prop);
    }

    return styles;
  }

  private countChildren(element: HTMLElement): number {
    return element.childElementCount;
  }
}

import { V4WidgetDetection } from '../extractor/v4-atomic-detector';
import { V4Tree, V4Node, V4Style } from '../types/v4-types';

export class V4TreeBuilder {
  private gcGenerator: GlobalClassesGenerator;
  private styleMerger: StyleMerger;

  constructor() {
    this.gcGenerator = new GlobalClassesGenerator();
    this.styleMerger = new StyleMerger();
  }

  buildTree(detections: V4WidgetDetection[]): V4Tree {
    const nodes: V4Node[] = [];
    const styles: Record<string, V4Style> = {};

    for (const detection of detections) {
      const node = this.buildNode(detection, styles);
      nodes.push(node);
    }

    // Generate global classes
    const gcPlan = this.gcGenerator.generate(nodes, styles);

    // Apply global classes to nodes
    this.applyGlobalClasses(nodes, gcPlan);

    return {
      elements: nodes,
      styles: Object.values(styles),
      globalClasses: gcPlan.classes,
      variables: gcPlan.variables,
    };
  }

  private buildNode(
    detection: V4WidgetDetection,
    styles: Record<string, V4Style>
  ): V4Node {
    const styleId = this.generateStyleId(detection.type);
    const style = this.createStyleFromDetection(detection, styleId);

    styles[styleId] = style;

    return {
      id: this.generateNodeId(),
      elType: detection.type,
      widgetType: detection.type,
      settings: {
        classes: {
          $$type: 'classes',
          value: [styleId],
        },
      },
      styles: {
        [styleId]: style,
      },
      elements: this.buildChildNodes(detection),
    };
  }

  private createStyleFromDetection(
    detection: V4WidgetDetection,
    styleId: string
  ): V4Style {
    return {
      id: styleId,
      type: 'class',
      label: `local-${styleId}`,
      variants: [
        {
          meta: {
            breakpoint: 'desktop',
            state: null,
          },
          props: this.convertStylesToProps(detection.styles),
          custom_css: null,
        },
      ],
    };
  }

  private convertStylesToProps(
    styles: Record<string, string>
  ): Record<string, any> {
    const props: Record<string, any> = {};

    for (const [key, value] of Object.entries(styles)) {
      // Convert CSS properties to V4 prop format
      const v4Prop = this.mapCssToV4Prop(key, value);
      if (v4Prop) {
        props[v4Prop.key] = v4Prop.value;
      }
    }

    return props;
  }

  private mapCssToV4Prop(
    cssKey: string,
    cssValue: string
  ): { key: string; value: any } | null {
    // Mapping logic from CSS to V4 props
    const mappings: Record<string, (value: string) => any> = {
      'background-color': (value) => ({
        $$type: 'color',
        value: this.parseColor(value),
      }),
      'font-family': (value) => ({
        $$type: 'font',
        value: { family: value },
      }),
      'font-size': (value) => ({
        $$type: 'size',
        value: this.parseSize(value),
      }),
      // Add more mappings...
    };

    const mapper = mappings[cssKey];
    if (!mapper) return null;

    return {
      key: this.cssToV4Key(cssKey),
      value: mapper(cssValue),
    };
  }

  private generateStyleId(widgetType: string): string {
    return `s-${widgetType}-${Date.now()}-${Math.floor(Math.random() * 1000)}`;
  }

  private generateNodeId(): string {
    return `n-${Date.now()}-${Math.floor(Math.random() * 1000)}`;
  }
}

import { GuardContext, GuardResult } from './types';

export class GuardsSystem {
  private guards: Map<string, Guard> = new Map();

  registerGuard(guard: Guard): void {
    this.guards.set(guard.id, guard);
  }

  async runAllGuards(tree: any, context: GuardContext): Promise<GuardResult[]> {
    const results: GuardResult[] = [];

    for (const [id, guard] of this.guards) {
      try {
        const result = guard.validate({ ...context, tree });
        results.push(result);

        if (result.severity === 'error' && result.errors.length > 0) {
          context.reportError(`${id}: ${result.errors.join(', ')}`);
        }

        if (result.warnings.length > 0) {
          context.reportWarning(`${id}: ${result.warnings.join(', ')}`);
        }
      } catch (error) {
        context.reportError(`Guard ${id} failed: ${error.message}`);
        results.push({
          guardId: id,
          severity: 'error',
          errors: [`Guard execution failed: ${error.message}`],
          warnings: [],
          score: 0,
        });
      }
    }

    return results;
  }

  calculateOverallScore(results: GuardResult[]): number {
    if (results.length === 0) return 0;

    const totalScore = results.reduce((sum, result) => sum + result.score, 0);
    return totalScore / results.length;
  }
}

```typescript
// src/orchestrator/diff-engine.ts
import { V4Tree, V3Tree } from '../types';

export class DiffEngine {
  calculateDiff(
    oldTree: V4Tree | V3Tree,
    newTree: V4Tree | V3Tree,
    options: { threshold?: number } = {}
  ): DiffResult {
    const { threshold = 0.1 } = options;

    const changes: Change[] = [];
    const additions: Addition[] = [];
    const deletions: Deletion[] = [];

    // Compare tree structures
    this.compareNodes(oldTree.elements, newTree.elements, changes, additions, deletions);

    // Compare styles
    this.compareStyles(oldTree.styles, newTree.styles, changes);

    // Compare global classes
    this.compareGlobalClasses(
      oldTree.globalClasses || [],
      newTree.globalClasses || [],
      changes
    );

    return {
      changes,
      additions,
      deletions,
      summary: {
        totalChanges: changes.length,
        totalAdditions: additions.length,
        totalDeletions: deletions.length,
        diffScore: this.calculateDiffScore(changes, additions, deletions),
      },
    };
  }

  private compareNodes(
    oldNodes: any[],
    newNodes: any[],
    changes: Change[],
    additions: Addition[],
    deletions: Deletion[]
  ): void {
    const oldMap = new Map(oldNodes.map((n) => [n.id, n]));
    const newMap = new Map(newNodes.map((n) => [n.id, n]));

    // Find deletions
    for (const [id, oldNode] of oldMap) {
      if (!newMap.has(id)) {
        deletions.push({
          type: 'node',
          id,
          path: this.getNodePath(oldNode),
        });
      }
    }

    // Find additions
    for (const [id, newNode] of newMap) {
      if (!oldMap.has(id)) {
        additions.push({
          type: 'node',
          node: newNode,
          path: this.getNodePath(newNode),
        });
      }
    }

    // Find changes
    for (const [id, newNode] of newMap) {
      const oldNode = oldMap.get(id);
      if (oldNode) {
        const nodeChanges = this.compareNode(oldNode, newNode);
        if (nodeChanges.length > 0) {
          changes.push({
            type: 'node',
            id,
            path: this.getNodePath(newNode),
            changes: nodeChanges,
          });
        }
      }
    }

    // Recursively compare children
    for (const [id, newNode] of newMap) {
      const oldNode = oldMap.get(id);
      if (oldNode && newNode.elements) {
        this.compareNodes(
          oldNode.elements || [],
          newNode.elements,
          changes,
          additions,
          deletions
        );
      }
    }
  }

  private compareNode(oldNode: any, newNode: any): NodeChange[] {
    const changes: NodeChange[] = [];

    // Compare settings
    if (JSON.stringify(oldNode.settings) !== JSON.stringify(newNode.settings)) {
      changes.push({
        property: 'settings',
        oldValue: oldNode.settings,
        newValue: newNode.settings,
      });
    }

    // Compare styles
    if (JSON.stringify(oldNode.styles) !== JSON.stringify(newNode.styles)) {
      changes.push({
        property: 'styles',
        oldValue: oldNode.styles,
        newValue: newNode.styles,
      });
    }

    // Compare elType
    if (oldNode.elType !== newNode.elType) {
      changes.push({
        property: 'elType',
        oldValue: oldNode.elType,
        newValue: newNode.elType,
      });
    }

    return changes;
  }

  private calculateDiffScore(
    changes: Change[],
    additions: Addition[],
    deletions: Deletion[]
  ): number {
    // Calculate diff score based on change magnitude
    const changeScore = changes.length * 0.5;
    const additionScore = additions.length * 0.3;
    const deletionScore = deletions.length * 0.2;

    return changeScore + additionScore + deletionScore;
  }
}

// src/orchestrator/incremental-builder.ts
import { DependencyGraphBuilder } from './dependency-graph';

export class IncrementalBuilder {
  private dependencyGraph: DependencyGraphBuilder;
  private batchSize: number;

  constructor(batchSize: number = 5) {
    this.dependencyGraph = new DependencyGraphBuilder();
    this.batchSize = batchSize;
  }

  async buildIncrementally(
    tree: V4Tree,
    options: { onProgress?: (progress: number) => void } = {}
  ): Promise<BuildResult> {
    const { onProgress } = options;

    // Build dependency graph
    this.buildDependencyGraph(tree);

    // Calculate build order
    const buildOrder = this.dependencyGraph.calculateBuildOrder();

    // Build in batches
    const results: BuildBatchResult[] = [];
    for (let i = 0; i < buildOrder.length; i += this.batchSize) {
      const batch = buildOrder.slice(i, i + this.batchSize);
      const result = await this.buildBatch(batch, tree);
      results.push(result);

      if (onProgress) {
        const progress = ((i + this.batchSize) / buildOrder.length) * 100;
        onProgress(Math.min(progress, 100));
      }
    }

    return {
      success: results.every((r) => r.success),
      batches: results,
      totalNodes: buildOrder.length,
    };
  }

  private buildDependencyGraph(tree: V4Tree): void {
    // Add all nodes to graph
    for (const node of tree.elements) {
      this.dependencyGraph.addNode(node.id, 'widget', node);
    }

    // Add dependencies based on parent-child relationships
    for (const node of tree.elements) {
      if (node.elements) {
        for (const childId of node.elements) {
          this.dependencyGraph.addDependency(node.id, childId);
        }
      }
    }

    // Add style dependencies
    for (const node of tree.elements) {
      if (node.settings?.classes?.value) {
        for (const styleId of node.settings.classes.value) {
          this.dependencyGraph.addDependency(node.id, `style:${styleId}`);
        }
      }
    }
  }

    nodeIds: string[],

        return [

            'errors' => $errors,
            'warnings' => $warnings,
            'score' => $this->calculateScore($errors, count($this->getAllChecks())),
            'checks_count' => count($this->getAllChecks()),
        ];
    }

- Building or editing V4 atomic pages

## V4 Best Practices

### 1. Atomic Structure
- Use `e-flexbox` or `e-div-block` for containers
- Avoid deeply nested structures (max 3 levels)
- Use `e-heading`, `e-button`, `e-image` for widgets
- Keep `elements` array flat when possible

### 2. Global Classes
- Create semantic class names (e.g., `gc-hero-section`)
- Use variants for responsive breakpoints
- Bind variables to class properties for consistency
- Reuse classes across multiple elements

### 3. Variables
- Define colors, typography, spacing as variables
- Use `$$type` envelope for all values
- Keep variable names kebab-case
- Group variables by type (color, typography, size)

### 4. Performance
- Minimize DOM depth
- Use global classes instead of inline styles
- Batch operations when possible
- Leverage components for reusable sections

## Common Issues & Solutions

| Issue | Cause | Solution |
|-------|-------|----------|
| `class_name_contains_spaces` | Hyphen in style ID | Rename to use underscores |
| `STYLE_CLASSES_BINDING FAIL` | Missing class reference | Add style ID to `classes.value` |
| `custom_css` crash | Plain string format | Use `{"raw": "..."}` format |
| Image doesn't load | `url: null` in image-src | Remove `url` key completely |
| Variable drift | Kit update changed IDs | Re-export design system & cross-validate |

## Validation Checklist
- [ ] All `$$type` envelopes are valid
- [ ] No visual properties in `settings`
- [ ] All style IDs referenced in `classes.value`
- [ ] Style IDs contain only `[a-z][a-z0-9_]*`

**2.1 Batch-Operations-System**
```php
// src/Abilities/V4/BatchOperations.php
<?php

class BatchOperations {
    private $batchSize = 50;
    private $maxExecutionTime = 30; // seconds

    public function batchUpdateContent($params) {
        $postIds = $params['post_ids'];
        $content = $params['content'];
        $batchSize = $params['batch_size'] ?? $this->batchSize;

        $results = [];
        $batches = array_chunk($postIds, $batchSize);

        foreach ($batches as $batchIndex => $batch) {
            $startTime = microtime(true);

            // Check execution time
            if (microtime(true) - $startTime > $this->maxExecutionTime) {
                $results[] = [
                    'batch_index' => $batchIndex,
                    'success' => false,
                    'error' => 'Execution time limit exceeded',
                    'post_ids' => $batch,
                ];
                break;
            }

            // Process batch
            $batchResults = $this->processBatch($batch, $content);

            $results[] = [
                'batch_index' => $batchIndex,
                'success' => true,
                'processed' => count($batch),
                'results' => $batchResults,
            ];
        }

        return [
            'success' => true,
            'total_batches' => count($batches),
            'processed_posts' => count($postIds),
            'results' => $results,
        ];
    }

    private function processBatch($postIds, $content) {
        $batchResults = [];

        foreach ($postIds as $postId) {
            try {
                // Update post content
                $result = $this->updatePostContent($postId, $content);
                $batchResults[] = [
                    'post_id' => $postId,
                    'success' => true,
                    'result' => $result,
                ];
            } catch (Exception $e) {
                $batchResults[] = [
                    'post_id' => $postId,
                    'success' => false,
                    'error' => $e->getMessage(),
                ];
            }
        }

        return $batchResults;
    }

    public function optimizeBatchOrder($postIds, $dependencies) {
        // Build dependency graph
        $graph = [];
        foreach ($postIds as $id) {
            $graph[$id] = [
                'dependencies' => $dependencies[$id] ?? [],
                'dependents' => [],
            ];
        }

        // Add reverse dependencies
        foreach ($postIds as $id) {
            foreach ($graph[$id]['dependencies'] as $dep) {
                if (isset($graph[$dep])) {
                    $graph[$dep]['dependents'][] = $id;
                }
            }
        }

        // Topological sort
        $sorted = [];
        $visited = [];
        $visiting = [];

        $visit = function($node) use (&$graph, &$sorted, &$visited, &$visiting) {
            if (isset($visited[$node])) return;
            if (isset($visiting[$node])) {
                throw new Exception("Circular dependency detected at $node");
            }

            $visiting[$node] = true;

            foreach ($graph[$node]['dependencies'] as $dep) {
                if (isset($graph[$dep])) {
                    $visit($dep);
                }
            }

            unset($visiting[$node]);
            $visited[$node] = true;
            $sorted[] = $node;
        };

        foreach ($postIds as $id) {
            $visit($id);
        }

        return $sorted;
    }
}

class DebugTools {

                'parameters' => [

                    'check_assets' => ['type' => 'boolean', 'default' => true],
                ],
                'handler' => [$this, 'analyzePerformance'],

            'v4-find-circular-dependencies' => [
                'name' => 'Find Circular Dependencies',
                'description' => 'Detect circular dependencies in V4 tree',
                'parameters' => [
                    'post_id' => ['type' => 'integer', 'required' => true],
                ],
                'handler' => [$this, 'findCircularDependencies'],
            ],
        ];
    }

    public function debugTreeStructure($params) {
        $postId = $params['post_id'];
        $maxDepth = $params['depth'] ?? 5;

        $content = get_post_meta($postId, '_elementor_data', true);
        $tree = json_decode($content, true);

        $analysis = [
            'total_nodes' => 0,
            'max_depth' => 0,
            'nodes_by_type' => [],
            'potential_issues' => [],
        ];

        $this->analyzeNode($tree, $analysis, 0, $maxDepth);

        return [
            'success' => true,
            'post_id' => $postId,
            'analysis' => $analysis,
        ];
    }

    private function analyzeNode($node, &$analysis, $currentDepth, $maxDepth) {
        if (!$node || $currentDepth > $maxDepth) return;

        $analysis['total_nodes']++;
        $analysis['max_depth'] = max($analysis['max_depth'], $currentDepth);

        // Count by type
        $type = $node['elType'] ?? 'unknown';
        $analysis['nodes_by_type'][$type] = ($analysis['nodes_by_type'][$type] ?? 0) + 1;

        // Check for potential issues
        if ($currentDepth > 3) {
            $analysis['potential_issues'][] = [
                'type' => 'deep_nesting',
                'node_id' => $node['id'] ?? null,
                'depth' => $currentDepth,
                'message' => 'Consider flattening structure',
            ];
        }

        // Check for empty containers
        if (empty($node['elements']) && in_array($node['elType'], ['e-flexbox', 'e-div-block'])) {
            $analysis['potential_issues'][] = [
                'type' => 'empty_container',
                'node_id' => $node['id'] ?? null,
                'message' => 'Empty container found',
            ];
        }

        // Recursively analyze children
        if (!empty($node['elements'])) {
            foreach ($node['elements'] as $child) {
                $this->analyzeNode($child, $analysis, $currentDepth + 1, $maxDepth);
            }
        }
    }
}

// src/Abilities/Pipeline/StateManager.php
<?php

class StateManager {
    private $stateDir;
    private $stateFile;

    public function __construct($stateDir = null) {
        $this->stateDir = $stateDir ?? WP_CONTENT_DIR . '/novamira-states';
        $this->ensureStateDir();
    }

    public function saveState($pipelineId, $state) {
        $this->stateFile = $this->stateDir . "/pipeline-{$pipelineId}.json";

        $state['timestamp'] = current_time('mysql');
        $state['version'] = '2.0.0';

        $result = file_put_contents(
            $this->stateFile,
            json_encode($state, JSON_PRETTY_PRINT)
        );

        return [
            'success' => $result !== false,
            'state_file' => $this->stateFile,
            'pipeline_id' => $pipelineId,
        ];
    }

    public function loadState($pipelineId) {
        $this->stateFile = $this->stateDir . "/pipeline-{$pipelineId}.json";

        if (!file_exists($this->stateFile)) {
            return [
                'success' => false,
                'error' => 'State file not found',
            ];
        }

        $content = file_get_contents($this->stateFile);
        $state = json_decode($content, true);

        return [
            'success' => true,
            'state' => $state,
            'pipeline_id' => $pipelineId,
        ];
    }

    public function cleanupOldStates($maxAge = 7 * DAY_IN_SECONDS) {
        $files = glob($this->stateDir . "/pipeline-*.json");
        $cleaned = 0;

        foreach ($files as $file) {
            if (filemtime($file) < time() - $maxAge) {
                unlink($file);
                $cleaned++;
            }
        }

        return [
            'success' => true,
            'cleaned_files' => $cleaned,
        ];
    }

    private function ensureStateDir() {
        if (!file_exists($this->stateDir)) {
            wp_mkdir_p($this->stateDir);
        }

        // Protect states directory
        if (!file_exists($this->stateDir . '/.htaccess')) {
            file_put_contents(
                $this->stateDir . '/.htaccess',
                "Order deny,allow\nDeny from all\n"
            );
        }
    }
}

class V4SecurityPolicies {

                'description' => 'Sanitize V4 tree content for security',
                'parameters' => [
                    'tree' => ['type' => 'object', 'required' => true],
                    'strict' => ['type' => 'boolean', 'default' => true],
                ],
                'handler' => [$this, 'sanitizeContent'],
            ],
        ];
    }

    public function validateSecurity($params) {
        $postId = $params['post_id'];

        $content = get_post_meta($postId, '_elementor_data', true);
        $tree = json_decode($content, true);

        $issues = [];

        // Check for XSS in custom CSS
        $cssIssues = $this->checkCustomCssForXss($tree);
        $issues = array_merge($issues, $cssIssues);

        // Check for malicious URLs
        $urlIssues = $this->checkUrlsForMalicious($tree);
        $issues = array_merge($issues, $urlIssues);

        // Check for insecure content
        $insecureIssues = $this->checkInsecureContent($tree);
        $issues = array_merge($issues, $insecureIssues);

        return [
            'success' => count($issues) === 0,
            'post_id' => $postId,
            'issues' => $issues,
            'checks_performed' => [
                'custom_css_xss',
                'malicious_urls',
                'insecure_content',
            ],
        ];
    }

    private function checkCustomCssForXss($tree) {
        $issues = [];

        $checkNode = function($node) use (&$checkNode, &$issues) {
            if (isset($node['styles'])) {
                foreach ($node['styles'] as $styleId => $style) {
                    if (isset($style['custom_css']) && is_array($style['custom_css'])) {
                        $css = $style['custom_css']['raw'] ?? '';
                        
                        // Check for dangerous patterns
                        $dangerousPatterns = [
                            '/javascript:/i',
                            '/on\w+\s*=/i',
                            '/<script/i',
                            '/eval\(/i',
                            '/expression\(/i',
                        ];

                        foreach ($dangerousPatterns as $pattern) {
                            if (preg_match($pattern, $css)) {
                                $issues[] = [
                                    'type' => 'xss_custom_css',
                                    'style_id' => $styleId,
                                    'pattern' => $pattern,
                                    'message' => 'Potentially dangerous JavaScript in custom CSS',
                                ];
                            }
                        }
                    }
                }
            }

            if (isset($node['elements'])) {
                foreach ($node['elements'] as $child) {
                    $checkNode($child);
                }
            }
        };

        $checkNode($tree);

        return $issues;
    }
}
